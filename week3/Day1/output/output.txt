    -- =========================================
    -- ROW_NUMBER() QUERIES
    -- =========================================
    
    -- 1. Row number based on highest salary
    
    SELECT
    emp_name,
    salary,
    ROW_NUMBER() OVER(ORDER BY salary DESC) AS row_num
    FROM employees;

    -- 2. Row number within department based on salary descending
    
    SELECT
    emp_name,
    department,
    salary,
    ROW_NUMBER() OVER(
    PARTITION BY department
    ORDER BY salary DESC
    ) AS row_num
