    -- =========================================

    -- 16. Rank orders within city based on order date
    
    SELECT
    order_id,
    city,
    order_date,
    RANK() OVER(
    PARTITION BY city
    ORDER BY order_date
    ) AS order_rank
    FROM orders;

    -- =========================================
    -- DENSE_RANK() QUERIES
    -- =========================================
    
    -- 17. Dense rank employees based on salary descending
    
    SELECT
    emp_name,
    salary,
    DENSE_RANK() OVER(
    ORDER BY salary DESC
