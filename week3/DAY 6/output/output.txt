    -- =========================================

    -- 19. Dense rank employees based on joining date
    
    SELECT
    emp_name,
    join_date,
    DENSE_RANK() OVER(
    ORDER BY join_date DESC
    ) AS dense_join_rank
    FROM employees;

    -- 21. Dense rank within city based on amount
    
    SELECT
    order_id,
    city,
    amount,
    DENSE_RANK() OVER(
    PARTITION BY city
