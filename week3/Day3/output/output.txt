    -- 9. Rank employees based on highest salary
    
    SELECT
    emp_name,
    salary,

    ORDER BY salary DESC

**Query #2**

    -- 10. Rank employees within department based on salary
    
    SELECT
    emp_name,
    department,
    salary,
    RANK() OVER(
    PARTITION BY department
    ORDER BY salary DESC
    ) AS dept_rank
    FROM employees;
