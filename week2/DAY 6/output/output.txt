    -- =========================================
    -- DUPLICATE ORDER IDS
    -- =========================================
    
    SELECT
    order_id,
    COUNT(*) AS duplicate_count
    FROM flipkart_raw_data
    GROUP BY order_id
    HAVING COUNT(*) > 1;
